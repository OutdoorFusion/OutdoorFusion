        function generateVisualization() {
    var chartType = document.getElementById('chart-type').value;
    var resultCount = document.getElementById('result-count').value;

    if (chartType === 'scatter') {
        generatePlot();
        return;
    }

    // Make a request to the appropriate route based on the selected chart type and result count
    var url = '/chart?type=' + chartType;

    if (chartType === 'pie') {
        var fieldA = document.getElementById('fieldA').value;
        url += '&field_a=' + fieldA;
    } else if (chartType === 'bar') {
        var fieldA = document.getElementById('fieldA').value;
        url += '&field_a=' + fieldA;
    }

    // Include the resultCount as a query parameter in the URL
    url += '&result_count=' + resultCount;

    Plotly.d3.json(url, function(error, data) {
        if (error) {
            console.log(error);
        }

        if (chartType === 'scatter') {
            Plotly.newPlot('plot', data.data, data.layout);
        } else if (chartType === 'pie') {
            var pieData = [{
                values: data.data[0].values,
                labels: data.data[0].labels,
                type: 'pie'
            }];
            var pieLayout = {
                title: ''
            };
            Plotly.newPlot('plot', pieData, pieLayout);
        } else if (chartType === 'bar') {
            var barData = [{
                x: data.data[0].x,
                y: data.data[0].y,
                type: 'bar'
            }];
            var barLayout = {
                title: ' '
            };
            Plotly.newPlot('plot', barData, barLayout);
        }
    });
}



  // Add event listener to the generate button for the plot
  document.getElementById('generate-btn').addEventListener('click', generateVisualization);
</script>
